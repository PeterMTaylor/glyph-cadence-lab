Object as Character
- Class Guild: holds lineage, rituals, and communal calibration.
- Class Drift: carries variation, resonance, and spectral shifts.
- Class Storyline: inherits from both, encoding logic as narrative.
class Guild:
    def calibrate(self, ritual):
        return "Guild performs " + ritual

class Drift(Guild):
    def reveal(self, glyph):
        return "Drift reveals meaning of " + glyph

class Storyline(Drift):
    def narrate(self, case):
        return "Given " + case + ", the Guild responds."
